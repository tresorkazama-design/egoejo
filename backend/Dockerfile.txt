FROM python:3.11-slim
# Assure que la sortie de Python est immédiate pour les logs
ENV PYTHONDONTWRITEBYTECODE 1
ENV PYTHONUNBUFFERED 1

# Crée le répertoire de travail
WORKDIR /app

# Installe les outils de construction et les dépendances PostgreSQL
RUN apt-get update && apt-get install -y build-essential libpq-dev

# Copie et installe les dépendances Python
COPY requirements.txt /app/
RUN pip install --upgrade pip && pip install -r requirements.txt

# Copie le reste du code
COPY . /app/

# Commande de démarrage (exécute les migrations et démarre Gunicorn)
# NOTE: Cette commande est écrasée par celle du docker-compose.yml, mais c'est le plan initial.
CMD ["gunicorn", "config.wsgi:application", "--bind", "0.0.0.0:8000", "--workers", "3"]
